def apply_pronunciation(text: str) -> str:
    for term, spoken in TTS_PRONUNCIATION.items():
        text = re.sub(rf"\b{re.escape(term)}\b", spoken, text)
    return text

def clean_text_for_tts(text: str) -> str:
    text = text.replace("\r\n", "\n").replace("\r", "\n")
    text = re.sub(r"[ \t]+", " ", text)
    text = re.sub(r"\n\s*\n\s*\n+", "\n\n", text)
    text = apply_pronunciation(text)
    return text.strip()


def get_image_files() -> list[Path]:
    if not IMAGE_FOLDER.exists():
        raise FileNotFoundError(f"Image folder not found: {IMAGE_FOLDER.resolve()}")

    image_files = sorted(
        [
            p for p in IMAGE_FOLDER.iterdir()
            if p.is_file() and p.suffix.lower() in IMAGE_EXTENSIONS
        ],
        key=lambda p: p.name.lower(),
    )

    if not image_files:
        raise RuntimeError(f"No images found in folder: {IMAGE_FOLDER.resolve()}")

    return image_files


def is_answer_image(image_path_or_name) -> bool:
    """True if the filename stem ends with _A (case-insensitive)."""
    return Path(image_path_or_name).stem.upper().endswith("_A")


def read_matching_text_file(image_path: Path) -> tuple[bool, str, str]:
    txt_path = image_path.with_suffix(".txt")

    if not txt_path.exists():
        reason = f"Matching txt file not found: {txt_path.name}"
        if SKIP_IMAGE_IF_TXT_MISSING:
            return False, "", reason
        return True, "This image contains visual information.", reason

    text = txt_path.read_text(encoding="utf-8", errors="ignore")
    text = clean_text_for_tts(text)

    if not text:
        reason = f"Matching txt file is empty: {txt_path.name}"
        if SKIP_IMAGE_IF_TXT_EMPTY:
            return False, "", reason
        return True, "This image contains visual information.", reason

    return True, text, ""


def load_bookend_text(
    text_file: Path,
    fallback: str,
    part_number: int,
    next_part: int,
) -> str:
    """Load intro/outro narration and fill {part}/{next_part}."""
    if text_file and text_file.exists():
        raw = text_file.read_text(encoding="utf-8", errors="ignore")
        raw = clean_text_for_tts(raw)
        if not raw:
            raw = fallback
    else:
        raw = fallback

    try:
        return raw.format(part=part_number, next_part=next_part)
    except (KeyError, IndexError, ValueError):
        return raw


def count_remaining_processable_images(
    image_files: list[Path],
    next_start_index: int,
) -> int:
    remaining_count = 0
    for remaining_image in image_files[next_start_index:]:
        text_ok, _, _ = read_matching_text_file(remaining_image)
        if text_ok:
            remaining_count += 1
    return remaining_count


def trim_white_border(img: Image.Image) -> Image.Image:
    if not TRIM_WHITE_BORDER:
        return img

    gray = img.convert("L")
    pixels = gray.load()
    width, height = gray.size

    left = width
    top = height
    right = -1
    bottom = -1

    for y in range(height):
        for x in range(width):
            if pixels[x, y] < WHITE_TRIM_THRESHOLD:
                if x < left:
                    left = x
                if x > right:
                    right = x
                if y < top:
                    top = y
                if y > bottom:
                    bottom = y

    if right == -1 or bottom == -1:
        return img

    left = max(0, left - WHITE_TRIM_MARGIN_PIXELS)
    top = max(0, top - WHITE_TRIM_MARGIN_PIXELS)
    right = min(width - 1, right + WHITE_TRIM_MARGIN_PIXELS)
    bottom = min(height - 1, bottom + WHITE_TRIM_MARGIN_PIXELS)

    cropped_width = right - left + 1
    cropped_height = bottom - top + 1

    if cropped_width < width * 0.20 or cropped_height < height * 0.20:
        return img

    return img.crop((left, top, right + 1, bottom + 1))


def resize_image_for_video(image_path: Path, output_path: Path):
    img = Image.open(image_path).convert("RGB")
    img = trim_white_border(img)

    target_width = int(VIDEO_WIDTH * IMAGE_SCREEN_FILL_RATIO)
    target_height = int(VIDEO_HEIGHT * IMAGE_SCREEN_FILL_RATIO)

    img.thumbnail((target_width, target_height), Image.Resampling.LANCZOS)

    canvas = Image.new("RGB", (VIDEO_WIDTH, VIDEO_HEIGHT), "white")
    x = (VIDEO_WIDTH - img.width) // 2
    y = (VIDEO_HEIGHT - img.height) // 2
    canvas.paste(img, (x, y))
    canvas.save(output_path, quality=98, optimize=True)


async def create_edge_tts_audio_async(text: str, audio_path: Path):
    communicate = edge_tts.Communicate(
        text=text,
        voice=EDGE_TTS_VOICE,
        rate=EDGE_TTS_RATE,
        volume=EDGE_TTS_VOLUME,

def create_tts_audio(text: str, audio_path: Path):
    asyncio.run(create_edge_tts_audio_async(text, audio_path))


def get_media_duration(media_path: Path) -> float:
    command = [
        "ffprobe", "-v", "error",
        "-show_entries", "format=duration",
        "-of", "json",
        str(media_path),
    ]
    result = subprocess.run(command, capture_output=True, text=True, check=True)
    data = json.loads(result.stdout)
    return float(data["format"]["duration"])


def build_atempo_filter(speed: float) -> list[str]:
    if speed <= 0:
        raise ValueError("AUDIO_SPEED must be greater than 0.")

    filters = []
    remaining_speed = speed

    while remaining_speed > 2.0:
        filters.append("atempo=2.0")
        remaining_speed = remaining_speed / 2.0
    while remaining_speed < 0.5:
        filters.append("atempo=0.5")
        remaining_speed = remaining_speed / 0.5
    if abs(remaining_speed - 1.0) > 0.001:
        filters.append(f"atempo={remaining_speed:.3f}")

    return filters


def create_video_segment(
    image_path: Path,
    audio_path: Path,
    segment_path: Path,
    is_question: bool = False,
) -> float:
    """
    Create a video segment from one image and one audio file.
    Question slides get a countdown timer during the pause after narration.
    Loudness normalization and voice polish are NOT applied here; they are
    applied once to the finished video for consistency.
    """
    original_audio_duration = get_media_duration(audio_path)
    speech_duration = original_audio_duration / AUDIO_SPEED

    timer_active = (
        is_question and SHOW_QUESTION_TIMER and QUESTION_TIMER_SECONDS > 0
    )

    tail_silence = float(QUESTION_TIMER_SECONDS) if timer_active else PAUSE_AFTER_SPEECH
    total_duration = speech_duration + tail_silence

    # ---- audio filters ----
    audio_filters = []
    audio_filters.extend(build_atempo_filter(AUDIO_SPEED))  # empty when AUDIO_SPEED == 1.0
    audio_filters.append(f"apad=pad_dur={tail_silence}")
    audio_filter_string = ",".join(audio_filters)

    # ---- video filters ----
    video_filters = [f"scale={VIDEO_WIDTH}:{VIDEO_HEIGHT}:flags=lanczos"]

    if timer_active:
        start = speech_duration
        font_opt = f"fontfile='{TIMER_FONT_FILE}':" if TIMER_FONT_FILE else ""

        video_filters.append(
            f"drawtext={font_opt}text='{TIMER_TEXT}':"
            f"enable='gte(t,{start:.3f})':"
            "x=(w-text_w)/2:y=h-270:fontsize=54:fontcolor=yellow:"
            "box=1:boxcolor=black@0.5:boxborderw=18"
        )
        video_filters.append(
            f"drawtext={font_opt}"
            f"text='%{{eif:max(0,ceil({total_duration:.3f}-t)):d}}':"
            f"enable='gte(t,{start:.3f})':"
            "x=(w-text_w)/2:y=h-200:fontsize=120:fontcolor=white:"
            "box=1:boxcolor=black@0.55:boxborderw=28"
        )
        if SHOW_TIMER_PROGRESS_BAR:
            video_filters.append(
                "drawbox=x=0:y=ih-18:"
                f"w='iw*max(0,({total_duration:.3f}-t)/{float(QUESTION_TIMER_SECONDS):.3f})':"
                "h=18:color=orange@0.9:t=fill:"
                f"enable='gte(t,{start:.3f})'"
            )

    video_filters.append("format=yuv420p")
    video_filter_string = ",".join(video_filters)

    command = [
        "ffmpeg", "-y",
        "-loop", "1",
        "-i", str(image_path),
        "-i", str(audio_path),
        "-t", f"{total_duration:.3f}",
        "-r", str(FPS),
        "-vf", video_filter_string,
        "-af", audio_filter_string,
        "-c:v", "libx264",
        "-preset", "slow",
        "-tune", "stillimage",
        "-crf", "16",
        "-pix_fmt", "yuv420p",
        "-c:a", "aac",
        "-b:a", "192k",
        "-ar", str(AUDIO_SAMPLE_RATE),
        str(segment_path),
    ]
    subprocess.run(command, check=True)
    return total_duration


def build_bookend_segment(
    image_path: Path,
    text: str,
    silent_duration: float,
    segment_path: Path,
    audio_tmp: Path,
) -> float:
    """Intro/outro from a static image; narrate text or hold silently."""
    processed = segment_path.with_suffix(".prep.jpg")
    resize_image_for_video(image_path=image_path, output_path=processed)

    if text.strip():
        create_tts_audio(text=text, audio_path=audio_tmp)
        return create_video_segment(
            image_path=processed,
            audio_path=audio_tmp,
            segment_path=segment_path,
            is_question=False,
        )

    command = [
        "ffmpeg", "-y",
        "-loop", "1",
        "-i", str(processed),
        "-f", "lavfi",
        "-i", f"anullsrc=channel_layout=stereo:sample_rate={AUDIO_SAMPLE_RATE}",
        "-t", f"{silent_duration:.3f}",
        "-r", str(FPS),
        "-vf", f"scale={VIDEO_WIDTH}:{VIDEO_HEIGHT}:flags=lanczos,format=yuv420p",
        "-c:v", "libx264",
        "-preset", "slow",
        "-tune", "stillimage",
        "-crf", "16",
        "-pix_fmt", "yuv420p",
        "-c:a", "aac",
        "-b:a", "192k",
        "-ar", str(AUDIO_SAMPLE_RATE),
        str(segment_path),
    ]
    subprocess.run(command, check=True)
    return float(silent_duration)


def concatenate_with_transitions(
    segment_files: list[Path],
    voice_only_output: Path,
    transition: str = XFADE_TRANSITION,
    duration: float = XFADE_DURATION,
):
    """
    Join segments with xfade (video) + acrossfade (audio), then apply the
    voice-polish chain to the combined narration. Loudness normalization is
    done later, on the final render.
    """
    n = len(segment_files)
    if n == 0:
        return

    durations = [get_media_duration(s) for s in segment_files]

    inputs = []
    for s in segment_files:
        inputs += ["-i", str(s)]

    # Normalize every video input so xfade never complains about
    # mismatched fps / timebase / pixel format.
    norm_parts = []
    for i in range(n):
        norm_parts.append(f"[{i}:v]fps={FPS},format=yuv420p,settb=AVTB[nv{i}]")

    # Video crossfade chain.
    v_parts = []
    prev_v = "nv0"
    cum = durations[0]
    for i in range(1, n):
        offset = cum - i * duration
        out_v = f"vx{i}"
        v_parts.append(
            f"[{prev_v}][nv{i}]xfade=transition={transition}:"
            f"duration={duration}:offset={offset:.3f}[{out_v}]"
        )
        prev_v = out_v
        cum += durations[i]

    # Audio crossfade chain.
    a_parts = []
    prev_a = "0:a"
    for i in range(1, n):
        out_a = f"ax{i}"
        a_parts.append(f"[{prev_a}][{i}:a]acrossfade=d={duration}[{out_a}]")
        prev_a = out_a

    # Final voice-polish stage (always produces a labeled [aout] pad).
    final_audio_filter = (
        VOICE_POLISH_FILTERS
        if (APPLY_VOICE_POLISH and VOICE_POLISH_FILTERS.strip())
        else "anull"
    )
    a_parts.append(f"[{prev_a}]{final_audio_filter}[aout]")

    filter_complex = ";".join(norm_parts + v_parts + a_parts)

    command = [
        "ffmpeg", "-y",
        *inputs,
        "-filter_complex", filter_complex,
        "-map", f"[{prev_v}]",
        "-map", "[aout]",
        "-c:v", "libx264",
        "-preset", XFADE_ENCODE_PRESET,
        "-crf", str(XFADE_ENCODE_CRF),
        "-pix_fmt", "yuv420p",
        "-c:a", "aac",
        "-b:a", "192k",
        "-ar", str(AUDIO_SAMPLE_RATE),
        str(voice_only_output),
    ]
    subprocess.run(command, check=True)


def finalize_normalize(voice_only_video: Path, final_video: Path):
    """Apply a single final loudness pass (video copied, audio re-encoded)."""
    if not NORMALIZE_AUDIO:
        shutil.copy2(voice_only_video, final_video)
        return

    command = [
        "ffmpeg", "-y",
        "-i", str(voice_only_video),
        "-c:v", "copy",
        "-af", LOUDNORM_TARGET,
        "-c:a", "aac",
        "-b:a", "192k",
        "-ar", str(AUDIO_SAMPLE_RATE),
        str(final_video),
    ]
    subprocess.run(command, check=True)


def add_background_music(voice_only_video: Path, final_video: Path):
    """Loop music, duck it under the narration (sidechain), normalize final."""
    if not BACKGROUND_MUSIC.exists():
        print("  Background music file not found. Normalizing voice-only video.")
        finalize_normalize(voice_only_video, final_video)
        return

    norm_stage = LOUDNORM_TARGET if NORMALIZE_AUDIO else "anull"

    filter_complex = (
        f"[1:a]aresample={AUDIO_SAMPLE_RATE},aformat=channel_layouts=stereo,"
        f"volume={BACKGROUND_MUSIC_VOLUME}[bg];"
        f"[0:a][bg]sidechaincompress=threshold=0.03:ratio=8:attack=5:release=400[mix];"
        f"[mix]{norm_stage}[a]"
    )

    command = [
        "ffmpeg", "-y",
        "-i", str(voice_only_video),
        "-stream_loop", "-1",
        "-i", str(BACKGROUND_MUSIC),
        "-filter_complex", filter_complex,
        "-map", "0:v",
        "-map", "[a]",
        "-c:v", "copy",
        "-c:a", "aac",
        "-b:a", "192k",
        "-ar", str(AUDIO_SAMPLE_RATE),
        "-shortest",
        str(final_video),
    ]
    subprocess.run(command, check=True)


def save_video_part(
    segment_files: list[Path],
    image_names: list[str],
    part_number: int,
    total_duration_seconds: float,
    is_last_part: bool = False,
):
    if not segment_files:
        return

    final_video = OUTPUT_FOLDER / f"{part_number:02d}_{OUTPUT_VIDEO_BASENAME}.mp4"
    voice_only_video = WORK_FOLDER / f"{part_number:02d}_voice_only.mp4"

    start_image = image_names[0] if image_names else "N/A"
    end_image = image_names[-1] if image_names else "N/A"
    next_part = part_number + 1

    full_segment_list = list(segment_files)

    if USE_INTRO and INTRO_IMAGE.exists():
        intro_seg = SEGMENT_FOLDER / f"intro_{part_number:02d}.mp4"
        intro_audio = AUDIO_FOLDER / f"intro_{part_number:02d}.mp3"
        intro_text = load_bookend_text(
            text_file=INTRO_TEXT_FILE,
            fallback=INTRO_TEXT_FALLBACK,
            part_number=part_number,
            next_part=next_part,
        )
        build_bookend_segment(
            image_path=INTRO_IMAGE,
            text=intro_text,
            silent_duration=INTRO_SILENT_DURATION,
            segment_path=intro_seg,
            audio_tmp=intro_audio,
        )
        full_segment_list.insert(0, intro_seg)
    elif USE_INTRO:
        print(f"  Intro image not found: {INTRO_IMAGE}. Skipping intro.")

    if USE_OUTRO and OUTRO_IMAGE.exists():
        if is_last_part:
            outro_text = load_bookend_text(
                text_file=OUTRO_TEXT_FILE_LAST,
                fallback=OUTRO_TEXT_LAST_FALLBACK,
                part_number=part_number,
                next_part=next_part,
            )
        else:
            outro_text = load_bookend_text(
                text_file=OUTRO_TEXT_FILE,
                fallback=OUTRO_TEXT_FALLBACK,
                part_number=part_number,
                next_part=next_part,
            )

        outro_seg = SEGMENT_FOLDER / f"outro_{part_number:02d}.mp4"
        outro_audio = AUDIO_FOLDER / f"outro_{part_number:02d}.mp3"
        build_bookend_segment(
            image_path=OUTRO_IMAGE,
            text=outro_text,
            silent_duration=OUTRO_SILENT_DURATION,
            segment_path=outro_seg,
            audio_tmp=outro_audio,
        )
        full_segment_list.append(outro_seg)
    elif USE_OUTRO:
        print(f"  Outro image not found: {OUTRO_IMAGE}. Skipping outro.")

    print()
    print("=" * 70)
    print(f"Creating video part {part_number:02d}"
          + ("  (final part)" if is_last_part else ""))
    print(f"Question/answer segments: {len(segment_files)} (+ intro/outro bookends)")
    print(f"Total segments to blend: {len(full_segment_list)}")
    print(f"Approx duration (before transitions): {total_duration_seconds / 60:.2f} minutes")
    print(f"Start image: {start_image}")
    print(f"End image: {end_image}")
    print(f"Transition: {XFADE_TRANSITION}  ({XFADE_DURATION}s)")
    print(f"Background music: {'On' if USE_BACKGROUND_MUSIC else 'Off'}")
    print(f"Output: {final_video}")
    print("=" * 70)

    concatenate_with_transitions(
        segment_files=full_segment_list,
        voice_only_output=voice_only_video,
    )

    if USE_BACKGROUND_MUSIC:
        add_background_music(voice_only_video, final_video)
    else:
        finalize_normalize(voice_only_video, final_video)

    log_video_saved(video_name=final_video.name, image_names=image_names)

    print(f"Saved: {final_video}")
    print(f"Logged video range: {start_image}, {end_image}")


def main():
    target_duration_seconds = TARGET_VIDEO_DURATION_MINUTES * 60

    prepare_folders()

    processed_count = 0
    skipped_count = 0
    failed_count = 0

    try:
        image_files = get_image_files()

        print(f"Found {len(image_files)} image(s) in: {IMAGE_FOLDER.resolve()}")
        print(f"Output folder: {OUTPUT_FOLDER.resolve()}")
        print(f"Log file: {LOG_FILE.resolve()}")
        print(f"Target video duration: {TARGET_VIDEO_DURATION_MINUTES} minutes")
        print("-" * 70)
        print("Edge TTS Settings")
        print(f"Voice: {EDGE_TTS_VOICE}")
        print(f"Rate: {EDGE_TTS_RATE}   Extra speed: {AUDIO_SPEED}x")
        print(f"Volume: {EDGE_TTS_VOLUME}   Pitch: {EDGE_TTS_PITCH}")
        print("-" * 70)
        print(f"Sample rate: {AUDIO_SAMPLE_RATE} Hz")
        print(f"Final loudness: {'Enabled (' + LOUDNORM_TARGET + ')' if NORMALIZE_AUDIO else 'Disabled'}")
        print(f"Voice polish: {'Enabled' if APPLY_VOICE_POLISH else 'Disabled'}")
        print(f"Pronunciation fixes: {len(TTS_PRONUNCIATION)} terms")
        print(f"Pause after speech: {PAUSE_AFTER_SPEECH} seconds")
        print(f"Question timer: {'Enabled' if SHOW_QUESTION_TIMER else 'Disabled'} "
              f"({QUESTION_TIMER_SECONDS}s)")
        print(f"Intro: {'On' if USE_INTRO else 'Off'}  Outro: {'On' if USE_OUTRO else 'Off'}")
        print(f"Transition: {XFADE_TRANSITION} ({XFADE_DURATION}s), "
              f"preset={XFADE_ENCODE_PRESET}, crf={XFADE_ENCODE_CRF}")
        print(f"Background music: {'On' if USE_BACKGROUND_MUSIC else 'Off'} "
              f"(volume {BACKGROUND_MUSIC_VOLUME}, ducked)")
        print("Split rule: save only after target duration AND after an _A image.")
        print("-" * 70)

        current_video_segments = []
        current_video_image_names = []
        current_video_duration = 0.0
        video_part_number = 1

        for index, image_path in enumerate(image_files, start=1):
            print(f"Processing {index}/{len(image_files)}: {image_path.name}")

            try:
                text_ok, text, text_reason = read_matching_text_file(image_path)

                if not text_ok:
                    skipped_count += 1
                    log_issue("SKIPPED", image_path.name, text_reason)
                    print(f"  Skipped: {text_reason}")
                    print("-" * 70)
                    continue

                if text_reason:
                    print(f"  Warning: {text_reason}")

                processed_image = PROCESSED_IMAGES / f"{index:05d}.jpg"
                audio_file = AUDIO_FOLDER / f"{index:05d}.mp3"
                segment_file = SEGMENT_FOLDER / f"{index:05d}.mp4"

                print(f"  Text length: {len(text)} characters")
                print(f"  Text file: {image_path.with_suffix('.txt').name}")

                resize_image_for_video(image_path=image_path, output_path=processed_image)
                create_tts_audio(text=text, audio_path=audio_file)

                current_image_is_answer = is_answer_image(image_path)

                segment_duration = create_video_segment(
                    image_path=processed_image,
                    audio_path=audio_file,
                    segment_path=segment_file,
                    is_question=not current_image_is_answer,
                )

                current_video_segments.append(segment_file)
                current_video_image_names.append(image_path.name)
                current_video_duration += segment_duration
                processed_count += 1

                print(f"  Segment duration: {segment_duration / 60:.2f} minutes")
                print(f"  Current video duration: {current_video_duration / 60:.2f} minutes")
                print(f"  Ends with _A image: {'Yes' if current_image_is_answer else 'No'}")

                if current_video_duration >= target_duration_seconds:
                    if current_image_is_answer:
                        remaining_processable_images = count_remaining_processable_images(
                            image_files=image_files,
                            next_start_index=index,
                        )
                        print("  Remaining processable image(s) after this image: "
                              f"{remaining_processable_images}")

                        if remaining_processable_images <= MERGE_FINAL_REMAINING_IMAGE_LIMIT:
                            print("  Target reached at _A, but remaining images are small.")
                            print(f"  Because remaining <= {MERGE_FINAL_REMAINING_IMAGE_LIMIT}, "
                                  "they will be included in the current video.")
                            print("-" * 70)
                            continue

                        print("  Target reached and current image is _A. Saving video part.")
                        print("-" * 70)

                        save_video_part(
                            segment_files=current_video_segments,
                            image_names=current_video_image_names,
                            part_number=video_part_number,
                            total_duration_seconds=current_video_duration,
                            is_last_part=False,
                        )

                        video_part_number += 1
                        current_video_segments = []
                        current_video_image_names = []
                        current_video_duration = 0.0

                    else:
                        print("  Target reached, but current image is not _A. Continuing.")
                        print("-" * 70)
                else:
                    print("-" * 70)

            except Exception as e:
                failed_count += 1
                reason = str(e).splitlines()[0]
                log_issue("FAILED", image_path.name, reason)
                print(f"  Failed: {reason}")
                print("-" * 70)
                continue

        if current_video_segments:
            last_image_name = current_video_image_names[-1]

            if not is_answer_image(last_image_name):
                warning_reason = (
                    "Final remaining video does not end with _A because no later _A image was found."
                )
                log_issue("WARNING", last_image_name, warning_reason)
                print()
                print("WARNING:")
                print(f"  {warning_reason}")
                print(f"  Last image: {last_image_name}")

            save_video_part(
                segment_files=current_video_segments,
                image_names=current_video_image_names,
                part_number=video_part_number,
                total_duration_seconds=current_video_duration,
                is_last_part=True,
            )

        print()
        print("Done.")
        print(f"Videos saved in: {OUTPUT_FOLDER.resolve()}")
        print(f"Log saved at: {LOG_FILE.resolve()}")
        print("-" * 70)
        print(f"Processed images: {processed_count}")
        print(f"Skipped images: {skipped_count}")
        print(f"Failed images: {failed_count}")

    finally:
        cleanup_work_folder()
        print("Temporary working folder deleted.")


if __name__ == "__main__":
    main()
